export DB_USER="payment_ledger"

# Create or update database user
if gcloud sql users list --instance="$CLOUD_SQL_INSTANCE" --format="value(name)" --project="$GCP_PROJECT_ID" 2>/dev/null | grep -q "^${DB_USER}$"; then
  gcloud sql users set-password "$DB_USER" \
    --instance="$CLOUD_SQL_INSTANCE" \
    --password="$DB_PASSWORD" \
    --project="$GCP_PROJECT_ID"
else
  gcloud sql users create "$DB_USER" \
    --instance="$CLOUD_SQL_INSTANCE" \
    --password="$DB_PASSWORD" \
    --project="$GCP_PROJECT_ID"
fi

export CLOUD_SQL_INSTANCE_CONNECTION_NAME="$(

# Helper function to create or update secrets in Secret Manager
create_or_update_secret() {
  local name="$1"
  local val="$2"
  gcloud secrets describe "$name" --project="$GCP_PROJECT_ID" >/dev/null 2>&1 || \
    gcloud secrets create "$name" --replication-policy="automatic" --project="$GCP_PROJECT_ID"
  printf '%s' "$val" | gcloud secrets versions add "$name" --data-file=- --project="$GCP_PROJECT_ID"
}

create_or_update_secret "payment-ledger-postgres-user" "$DB_USER"
